var webpack = require("webpack");

function absdir(r){
  return path.resolve(__dirname, r) ;
}

function output_fn(name, dirname){
  dirname = dirname || 'dist';
  var dir = absdir( dirname) ;
  return { path: dir, filename: name, publicPath: "/" + dirname, library: 'wdf' };
}
